def indent(spaces=2)
  return self.undent(-spaces) if spaces < 0

  lines = self.split $/
  lines.each do |line|
    if line.blank?
      line.clear
    else
      line.insert(0, ' '*spaces)
    end
  end
  lines.join $/
end

def undent(spaces=2)
  return self.indent(-spaces) if spaces < 0

  lines = self.split $/
  lines.each do |line|
    line.slice!(0...spaces)
  end
  lines.join $/
end

def underscore!
  gsub! /::/, '/'
  gsub! /([A-Z]+)([A-Z][a-z])/, '\1_\2'
  gsub! /([a-z\d])([A-Z])/, '\1_\2'
  tr!   '-', '_'
  downcase!
  self
end
