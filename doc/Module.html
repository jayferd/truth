def enumerate_by(method)
  include Enumerable
  %w(
    each
    to_a
    include?
    first
    last
  ).each do |enum_method|
    module_eval "def \#{enum_method}(*args, &blk)\nself.\#{method}.\#{enum_method}(*args, &blk)\nend unless instance_methods.include? :\#{enum_method}\n", __FILE__, __LINE__
  end
end
