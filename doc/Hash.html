def map!
  self.keys.each do |k|
    new_k, new_v = yield(k, self.delete(k))
    self[new_k] = new_v
  end
  self
end

def rmerge!(hsh)
  hsh.each do |k,v|
    self[k] = v unless self.has_key? k
  end
  self
end
