def require_local(*filename)
  filename = File.join(filename)
  local_glob = File.expand_path(
    File.join(
      File.dirname(
        caller.first.split(':').first
      ),
      *filename.split('/')
    )
  )

  local_glob.unchomp!('.rb')

  Dir.glob(local_glob).each do |file|
    require file
  end
end

def Truth(version=nil, &blk)
  return Truth unless version || block_given?

  version ||= Truth.create_version
  config = Truth::Configuration.version(version)

  if blk
    Truth::Dsl::ConfigurationDsl.new(config).instance_eval(&blk)
  end

  config
end
