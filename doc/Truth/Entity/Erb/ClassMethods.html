def erb(type)
  type = type.to_sym

  erbs[type] ||= begin
    template = File.read(File.join(
      Truth::DIRS[:lib],
      'truth',
      'templates',
      self.name.demodulize.underscore,
      "#{type}.#{subtypes[type]}.erb"
    ))

    Erubis::Eruby.new(template)
  end
end
