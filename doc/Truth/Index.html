def add(obj)
  return self if self.include? name_of(obj)

  hook_wrap :add, obj do
    begin
      name = name_of(obj)
      insert_sorted(obj)
      hashed[name] = obj
      obj
    rescue Exception => e
      delete(name)
      raise e
    end
  end

  self
end

def delete(name)
  hook_wrap :delete, name do

  end

def remove(obj)
  hashed.each do |k,v|
    hashed.delete(k) if v == obj
  end
  list.reject! { |el| el == obj }
end
